Plotly = require("https://cdn.plot.ly/plotly-latest.min.js")

{
  const dataEvasao = Array.from(
    d3.rollup(
      dados_filtrados,
      v => d3.sum(v, d => d.QTDE),
      d => d.FORMA_EVASAO
    ),
    ([tipo, total]) => ({ tipo, total })
  )
  .filter(d => d.total > 0)
  .sort((a, b) => d3.descending(a.total, b.total));

  // Preparar dados para o Plotly
  const evasaoAno = [{
    values: dataEvasao.map(d => d.total),
    labels: dataEvasao.map(d => d.tipo),
    textinfo: 'none',
    direction: 'clockwise',
    hole: .6,
    marker: {
      colors: d3.quantize(t => d3.interpolateGreys(t * 0.8 + 0.1), dataEvasao.length).reverse(),
      line: {
    color: 'white',
    width: 2,
      },  
    },
    type: 'pie',
    name: 'Formas de Evasão',
  }];

  const layout = {
      legend: {
      y: 0.5,
      x: 1.1,
            },
      font: {
    family: 'Poppins',
    size: 14,
            },
      margin: {
    b: 20,
    r: 20,
    l: 20,
    t: 50,
            },

  };

  const config = { 
    responsive: true,
    staticPlot: false,
    displaylogo: false,
  };

  const div = DOM.element('plot-container');
  Plotly.newPlot(div, evasaoAno, layout, config);
  return div;
}
